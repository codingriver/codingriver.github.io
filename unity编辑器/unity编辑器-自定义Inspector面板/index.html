    /// еҰӮжһңжҳҜжһҡдёҫ  еҲҷжҜҸдёӘеҖј+1еҗҺжҳҜеҜ№еә”зҡ„жһҡдёҫ
    /// 
    /// </summary>

    /// <param name="arrLen">ж•°з»„ж•°йҮҸпјҲеҰӮжһңжҳҜжһҡдёҫеҲҷжҳҜжһҡдёҫзҡ„ж•°йҮҸпјҢжҺ’йҷӨжһҡдёҫдёә0зҡ„еҖјпјү</param>
    /// <returns></returns>
    public static List<int> ResolveMixed(int mixedValue, int arrLen)
    {
        List<int> ls = new List<int>();
        //е…ЁйҖүж—¶еҖјдёә-1
        if (mixedValue == -1)
        {
            for (int i = 0; i < arrLen; i++)
                ls.Add(i);

            return ls;
        }

        for (int i = 0; i < 32; i++)
        {
            if ((mixedValue & 1) == 1) ls.Add(i);

            mixedValue = mixedValue >> 1;
        }
        return ls;
    }
